## Wikidata API

async function getElementInfo(qid: string) {
  const params = new URLSearchParams({
    action: 'wbgetentities',
    ids: qid,
    props: 'labels|descriptions|sitelinks',
    languages: 'ru|en',
    format: 'json',
    origin: '*'
  });
  
  const response = await fetch(`${WIKIDATA_API}?${params}`);
  const data = await response.json();
  
  return {
    name: data.entities[qid].labels.ru?.value,
    description: data.entities[qid].descriptions.ru?.value,
    wikiUrl: data.entities[qid].sitelinks.ruwiki?.url
  };
}
